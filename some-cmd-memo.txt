
# Login to Heroku
heroku login

# Review Heroku config for specified application, such as DATABASE_URL
heroku config

# Review PostgreSQL instance in Heroku for specified application
heroku pg --app <app name>

# instantiate PostgreSQL
heroku addons:create heroku-postgresql --app <app name>

# Connect to PostgreSQL server 
heroku pg:psql --app <app name>
for example:
  	heroku pg:psql --app nccu-cicd-demo-gh-1123-2
	heroku pg:psql --app nccu-cicd-demo-gh-1123-prod-2

# attach existing PstgreSQL DB from another app
heroku addons:attach <another app name>::DATABASE --app <new-app name>
